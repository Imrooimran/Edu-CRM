Apps Script Code 
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // ── Students ──
    var s1 = ss.getSheetByName('Students') || ss.insertSheet('Students');
    s1.clearContents();
    s1.appendRow(['Name','Phone','Email','Course','Batch','Stage','Source',
                  'Fees (INR)','Paid (INR)','Balance (INR)','Enrolled','Follow-up','Notes']);
    data.students.forEach(function(r) {
      s1.appendRow([r.name, r.phone, r.email||'', r.course, r.batch||'',
                    r.stage, r.source, r.fee, r.paid, r.fee - r.paid,
                    r.enrolled||'', r.followup||'', r.notes||'']);
    });

    // ── Clients ──
    var s2 = ss.getSheetByName('Clients') || ss.insertSheet('Clients');
    s2.clearContents();
    s2.appendRow(['Company','Contact','Email','Phone','Course','Team Size',
                  'Contract (INR)','Stage','Follow-up','Notes']);
    data.clients.forEach(function(r) {
      s2.appendRow([r.company, r.contact, r.email||'', r.phone||'',
                    r.course, r.teamSize||0, r.contract,
                    r.stage, r.followup||'', r.notes||'']);
    });

    return ContentService
      .createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:'ready'}))
    .setMimeType(ContentService.MimeType.JSON);
}
